cmake_minimum_required(VERSION 3.20)
project(enlightenedc
    VERSION 0.0.1
    LANGUAGES CXX
    DESCRIPTION "An LLVM-powered HolyC compiler"
)

# 1. Setting various requirements and variables

set(CMAKE_EXPORT_COMPILE_COMMANDS True)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED True)
# Use Clang as our default compiler, we are an LLVM project after all. :)
set(CMAKE_C_COMPILER clang)
set(CMAKE_CXX_COMPILER clang++)
add_compile_options("-O2")

set(SRC_DIR "${CMAKE_SOURCE_DIR}/src")
set(INCLUDE_DIR "${CMAKE_SOURCE_DIR}/include")

# option to toggle clang-tidy during compilation, on by default
option(ENABLE_CLANG_TIDY "Enable clang-tidy during compilation" ON)

# 2. Finding our dependencies

find_package(BISON 3.8 REQUIRED)
find_package(FLEX 2.6 REQUIRED)
find_package(LLVM REQUIRED CONFIG)

message(STATUS "Found LLVM ${LLVM_PACKAGE_VERSION}")
message(STATUS "Using LLVMConfig.cmake in: ${LLVM_DIR}")
separate_arguments(LLVM_DEFINITIONS_LIST NATIVE_COMMAND ${LLVM_DEFINITIONS})
add_definitions(${LLVM_DEFINITIONS_LIST})

# 3. Define Flex/Bison targets

# define and create a directory for code generated by Flex/Bison.
set(BISON_OUTPUT_DIR "${CMAKE_BINARY_DIR}/gen")
file(MAKE_DIRECTORY ${BISON_OUTPUT_DIR})

bison_target(
    parser
    "${SRC_DIR}/frontend/parser.yy"
    "${BISON_OUTPUT_DIR}/parser.cpp"
    DEFINES_FILE "${BISON_OUTPUT_DIR}/parser.hpp"
    VERBOSE REPORT_FILE "${CMAKE_BINARY_DIR}/parser.output"
)

flex_target(
    scanner
    "${SRC_DIR}/frontend/lexer.l"
    "${BISON_OUTPUT_DIR}/lexer.cpp"
)

add_flex_bison_dependency(scanner parser)

# 4. Collect source files
file(GLOB_RECURSE AST_SRC "${SRC_DIR}/ast/*.cpp")
file(GLOB_RECURSE PREPROC_SRC "${SRC_DIR}/preproc/*.cpp")
file(GLOB_RECURSE DRIVER_SRC "${SRC_DIR}/driver/*.cpp")
file(GLOB_RECURSE FRONTEND_SRC "${SRC_DIR}/frontend/*.cpp")
file(GLOB_RECURSE SEMANTICS_SRC "${SRC_DIR}/semantics/*.cpp")
file(GLOB_RECURSE CODEGEN_SRC "${SRC_DIR}/codegen/*.cpp")
file(GLOB_RECURSE EVAL_SRC "${SRC_DIR}/eval/*.cpp")
file(GLOB         ECC_SRC "${SRC_DIR}/*.cpp")

file(GLOB_RECURSE HEADERS "${INCLUDE_DIR}/*.hpp")

# All the files we wrote ourselves.
list(APPEND ALL_WRITTEN_FILES 
    ${AST_SRC} 
    ${PREPROC_SRC} 
    ${DRIVER_SRC} 
    ${FRONTEND_SRC}
    ${SEMANTICS_SRC}
    ${CODEGEN_SRC}
    ${EVAL_SRC}
    ${ECC_SRC}
)

# Add the generated files
list(APPEND ALL_SOURCE_FILES
    ${ALL_WRITTEN_FILES}
    ${BISON_parser_OUTPUTS} 
    ${FLEX_scanner_OUTPUTS}
)

set_source_files_properties(${ALL_WRITTEN_FILES} 
    PROPERTIES 
    COMPILE_OPTIONS "-Wall;-Wextra;-Wpedantic"
)

# 5. Create our final targets

# Create a static library that holds all our core functionality
add_library(libecc STATIC ${ALL_SOURCE_FILES})
set_target_properties(libecc PROPERTIES
    PREFIX ""
    OUTPUT_NAME "libecc"
    SUFFIX ".a"
)

# Create an executable from our main.cpp driver
add_executable(ecc "${SRC_DIR}/main.cpp")

# 6. Setup include paths
target_include_directories(libecc PRIVATE ${SRC_DIR} ${BISON_OUTPUT_DIR})
target_include_directories(libecc PUBLIC ${INCLUDE_DIR})
target_include_directories(ecc PUBLIC ${INCLUDE_DIR})

# 7. Link our targets and their dependencies
llvm_map_components_to_libnames(llvm_libs
    core
    executionengine
    orcjit
    native
    support
)

message(STATUS "LLVM libraries: ${LLVM_LIBRARY_DIRS}")
# Link LLVM to our library target
target_link_libraries(libecc PRIVATE LLVM)
# Link our library target to our executable
target_link_libraries(ecc PRIVATE libecc)

# 8. Set clang-tidy and clang-format
find_program(CLANG_TIDY_EXE NAMES "clang-tidy")
if(CLANG_TIDY_EXE)
    if (ENABLE_CLANG_TIDY)
        message(STATUS "clang-tidy found: ${CLANG_TIDY_EXE}")
        set_target_properties(ecc PROPERTIES CXX_CLANG_TIDY ${CLANG_TIDY_EXE})
    endif()
else()
    message(WARNING "clang-tidy not found")
endif()

find_program(CLANG_FORMAT_EXE NAMES "clang-format")
if(CLANG_FORMAT_EXE)
    add_custom_target(format
        COMMAND ${CLANG_FORMAT_EXE} -i ${ALL_WRITTEN_FILES} ${HEADERS}
        COMMENT "Format code with clang-format"
    )
else()
    message(WARNING "clang-format not found, `format` target is unavailable")
endif()

# 9. Add precompiled headers
target_precompile_headers(ecc PRIVATE 
    <vector> 
    <string>
    <memory>
    <unordered_map>
    <variant>
    <llvm/ADT/APFloat.h>
    <llvm/ADT/STLExtras.h>
    <llvm/IR/BasicBlock.h>
    <llvm/IR/Constants.h>
    <llvm/IR/DerivedTypes.h>
    <llvm/IR/Function.h>
    <llvm/IR/IRBuilder.h>
    <llvm/IR/Instructions.h>
    <llvm/IR/LLVMContext.h>
    <llvm/IR/LegacyPassManager.h>
    <llvm/IR/Module.h>
    <llvm/IR/Type.h>
    <llvm/IR/Verifier.h>
    <llvm/MC/TargetRegistry.h>
    <llvm/Support/FileSystem.h>
    <llvm/Support/TargetSelect.h>
    <llvm/Support/raw_ostream.h>
    <llvm/Target/TargetMachine.h>
    <llvm/Target/TargetOptions.h>
    <llvm/TargetParser/Host.h>
)
