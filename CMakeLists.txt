cmake_minimum_required(VERSION 3.20)
project(enlightenedc)

# 1. Setting various requirements and variables

set(CMAKE_EXPORT_COMPILE_COMMANDS True)
set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED True)
# Use Clang as our default compiler, we are an LLVM project after all. :)
set(CMAKE_C_COMPILER clang)
set(CMAKE_CXX_COMPILER clang++)

set(SRC_DIR "${CMAKE_SOURCE_DIR}/src")

# 2. Finding our dependencies

find_package(BISON 3.8 REQUIRED)
find_package(FLEX 2.6 REQUIRED)
find_package(LLVM REQUIRED CONFIG)

message(STATUS "Found LLVM ${LLVM_PACKAGE_VERSION}")
message(STATUS "Using LLVMConfig.cmake in: ${LLVM_DIR}")
separate_arguments(LLVM_DEFINITIONS_LIST NATIVE_COMMAND ${LLVM_DEFINITIONS})
add_definitions(${LLVM_DEFINITIONS_LIST})

# 3. Define Flex/Bison targets

# define and create a directory for code generated by Flex/Bison.
set(BISON_OUTPUT_DIR "${CMAKE_BINARY_DIR}/gen")
file(MAKE_DIRECTORY ${BISON_OUTPUT_DIR})

bison_target(
    parser
    "${SRC_DIR}/frontend/parser.yy"
    "${BISON_OUTPUT_DIR}/parser.cpp"
    DEFINES_FILE "${BISON_OUTPUT_DIR}/parser.hpp"
    VERBOSE REPORT_FILE "${CMAKE_BINARY_DIR}/parser.output"
)

flex_target(
    scanner
    "${SRC_DIR}/frontend/lexer.l"
    "${BISON_OUTPUT_DIR}/lexer.cpp"
)

add_flex_bison_dependency(scanner parser)

# 4. Collect source files
file(GLOB_RECURSE AST_SRC "${SRC_DIR}/ast/*.cpp")
file(GLOB_RECURSE PREPROC_SRC "${SRC_DIR}/preproc/*.cpp")
file(GLOB_RECURSE DRIVER_SRC "${SRC_DIR}/driver/*.cpp")
file(GLOB_RECURSE FRONTEND_SRC "${SRC_DIR}/frontend/*.cpp")
file(GLOB_RECURSE SEMANTICS_SRC "${SRC_DIR}/semantics/*.cpp")
file(GLOB_RECURSE CODEGEN_SRC "${SRC_DIR}/codegen/*.cpp")
file(GLOB_RECURSE EVAL_SRC "${SRC_DIR}/eval/*.cpp")
file(GLOB_RECURSE ECC_SRC "${SRC_DIR}/*.cpp")

# 5. Create our target executable
add_executable(ecc 
    ${AST_SRC}
    ${PREPROC_SRC}
    ${DRIVER_SRC}
    ${FRONTEND_SRC}
    ${SEMANTICS_SRC}
    ${CODEGEN_SRC} 
    ${EVAL_SRC}
    ${BISON_parser_OUTPUTS} 
    ${FLEX_scanner_OUTPUTS} 
    "${SRC_DIR}/ecc.cpp"
)

# 6. Setup include paths
target_include_directories(ecc PRIVATE ${SRC_DIR} ${BISON_OUTPUT_DIR})
target_precompile_headers(ecc PRIVATE 
    <vector> 
    <string>
    <memory>
    <unordered_map>
    <variant>
    <llvm/ADT/APFloat.h>
    <llvm/ADT/STLExtras.h>
    <llvm/IR/BasicBlock.h>
    <llvm/IR/Constants.h>
    <llvm/IR/DerivedTypes.h>
    <llvm/IR/Function.h>
    <llvm/IR/IRBuilder.h>
    <llvm/IR/Instructions.h>
    <llvm/IR/LLVMContext.h>
    <llvm/IR/LegacyPassManager.h>
    <llvm/IR/Module.h>
    <llvm/IR/Type.h>
    <llvm/IR/Verifier.h>
    <llvm/MC/TargetRegistry.h>
    <llvm/Support/FileSystem.h>
    <llvm/Support/TargetSelect.h>
    <llvm/Support/raw_ostream.h>
    <llvm/Target/TargetMachine.h>
    <llvm/Target/TargetOptions.h>
    <llvm/TargetParser/Host.h>
)

# 7. Link LLVM libraries
llvm_map_components_to_libnames(llvm_libs
    core
    executionengine
    orcjit
    native
    support
)
message(STATUS "LLVM libraries: ${LLVM_LIBRARY_DIRS}")
target_link_libraries(ecc PRIVATE LLVM)
